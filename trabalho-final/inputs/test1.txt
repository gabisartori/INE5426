def @soma(int a, int b) {
    int resultado;
    resultado = a + b;
    print "Resultado da soma = ";
    print resultado;
    return;
}

def @subtrai(int a, int b) {
    int resultado;
    resultado = a - b;
    print "Resultado da subtração = ";
    print resultado;
    return;
}

def @multiplica(int a, int b) {
    int resultado;
    resultado = a * b;
    print "Resultado da multiplicação = ";
    print resultado;
    return;
}

def @divide(int a, int b) {
    int resultado;
    if (b == 0) {
        print "Divisao por zero nao permitida";
    } else {
        resultado = a / b;
    }
    print "Resultado da divisão = ";
    print resultado;
    return;
}

def @fatorial(int n) {
    int resultado;
    resultado = 1;
    int i;
    for (i = 1; i <= n; i = i + 1) {
        resultado = resultado * i;
    }
    print "Resultado dp fatorial = ";
    print resultado;
    return;
}

def @potencia(int base, int expoente) {
    int resultado;
    int i;
    resultado = 1;
    for (i = 0; i < expoente; i = i + 1) {
        resultado = resultado * base;
    }
    print "Resultado da potência = ";
    print resultado;
    return;
}

def @modulo(int a, int b) {
    int resultado;
    resultado = a%b;
    print "Resultado do módulo = ";
    print resultado;
    return;
}
