def @setupGame() {
    heroHP = 30 + 5;
    heroPotions = 2;
    enemyHP = 35 + (5 * 1);
    isGameOver = 0;
    attackCount = 0;
    lastHit = 0;
    currentTurn = 1;
    return;
}

def @printIntro() {
    print "Um novo combate começa.";
    print "Você tem";
    print heroHP;
    print "pontos de vida e";
    print heroPotions;
    print "poções.";
    print "O inimigo tem";
    print enemyHP;
    print "de vida.";
    return;
}

def @playerTurn() {
    print "Turno:";
    print currentTurn;
    print "Escolha sua ação:";
    print "1 - Atacar";
    print "2 - Defender";
    print "3 - Usar poção";
    read turnChoice;

    if (turnChoice == 1) {
        damage = 7 + (currentTurn % 3);
        bonusDamage = 2 * (attackCount % 2);
        damage = damage + bonusDamage;
        enemyHP = enemyHP - damage;
        lastHit = damage;
        attackCount = attackCount + 1;
        print "Você ataca e causa:";
        print damage;
        print "Dano total causado até agora:";
        print attackCount * lastHit;
        print "Vida restante do inimigo:";
        print enemyHP;
    } else if (turnChoice == 2) {
        print "Você assume uma postura defensiva.";
    } else if (turnChoice == 3) {
        if (heroPotions != 0) {
            potionEffect = 5 + (heroPotions * 2);
            heroHP = heroHP + potionEffect;
            heroPotions = heroPotions - 1;
            print "Você bebe uma poção e recupera:";
            print potionEffect;
            print "Sua vida agora é:";
            print heroHP;
        }
        else {
            print "Você está sem poções!";
        }
    } else {
        print "Ação inválida. Você perde o turno.";
    }

    return;
}

def @enemyTurn() {
    print "O inimigo se prepara para atacar.";
    damage = 6;

    if (turnChoice == 2) {
        damage = damage / 2;
        damage = damage + (currentTurn % 2);
    }

    temp = damage;
    heroHP = heroHP - damage;

    print "Você recebe";
    print temp;
    print "de dano.";
    print "Sua vida agora é:";
    print heroHP;
    return;
}

def @checkEnd() {
    if (heroHP <= 0) {
        print "Você foi derrotado.";
        isGameOver = 1;
    } else if (enemyHP <= 0) {
        print "Você venceu!";
        isGameOver = 1;
    } else if (heroHP > 50) {
        print "Sua energia está impressionante!";
    } else if (heroHP < 10) {
        print "Você está gravemente ferido!";
    }
    return;
}

def @mainLoop() {
    int i;
    for (i = 0; i < 10; i = i + 1) {
        if (isGameOver == 1) {
            break;
        }

        i = @playerTurn();
        if (enemyHP > 0) {
            i = @enemyTurn();
        }

        i = @checkEnd();
        currentTurn = currentTurn + 1;
    }
    return;
}

def @main() {
    result = @setupGame();
    result = @printIntro();
    result = @mainLoop();
    return;
}
