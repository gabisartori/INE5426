SDD L-atribuída: Todos os atributos são sintetizados.

EXPA:
EXPRESSION -> NUMEXPRESSION
 EXPRESSION.ptr = NUMEXPRESSION.ptr
EXPRESSION -> NUMEXPRESSION OP_EXPRESSION NUMEXPRESSION
 EXPRESSION.ptr = OP_EXPRESSION(NUMEXPRESSION_1.ptr, NUMEXPRESSION_2.ptr)

NUMEXPRESSION -> TERM
 NUMEXPRESSION.ptr = TERM.ptr
NUMEXPRESSION -> NUMEXPRESSION OP_NUMEXPRESSION TERM
 NUMEXPRESSION_1.ptr = OP_NUMEXPRESSION(NUMEXPRESSION_2.ptr, TERM.ptr)

TERM -> UNARYEXPRESSION
 TERM.ptr = UNARYEXPRESSION.ptr
TERM -> TERM OP_TERM UNARYEXPRESSION
 TERM_1.ptr = OP_TERM(TERM_2.ptr, UNARYEXPRESSION.ptr)

UNARYEXPRESSION -> FACTOR
 UNARYEXPRESSION.ptr = FACTOR.ptr
UNARYEXPRESSION -> OP_NUMEXPRESSION FACTOR
 UNARYEXPRESSION.ptr = OP_NUMEXPRESSION(FACTOR.ptr)

FACTOR -> CONSTANT
 FACTOR.ptr = Constant(CONSTANT.val)
FACTOR -> EXPRESSION
 FACTOR.ptr = EXPRESSION.ptr
FACTOR -> LVALUE
 FACTOR.ptr = LVALUE.ptr

LVALUE -> id VAR_INDEX
 LVALUE.ptr = Identifier(id.val)
